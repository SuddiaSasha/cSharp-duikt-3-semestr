CREATE OR REPLACE FUNCTION get_even_id_rows()
RETURNS TABLE(id INT, name VARCHAR, price INT) AS $$
BEGIN
    RETURN QUERY SELECT clothes.id, clothes.name, clothes.price
    FROM clothes
    WHERE clothes.id % 2 = 0;
END;
$$ LANGUAGE plpgsql;
